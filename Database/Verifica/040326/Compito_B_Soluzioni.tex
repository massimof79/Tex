	\section*{Soluzioni -- Parte A (Query SQL)}
	% ============================================================
	
	\begin{enumerate}
		
		% ------ 1 ------
		\item \textbf{Pizze ordinate per categoria e nome}
		\begin{lstlisting}
			SELECT nome, prezzo, categoria
			FROM PIZZE
			ORDER BY categoria, nome;
		\end{lstlisting}
		
		% ------ 2 ------
		\item \textbf{Clienti con indirizzo contenente ``Via''}
		\begin{lstlisting}
			SELECT id_cliente, cognome, nome, indirizzo
			FROM CLIENTI
			WHERE indirizzo LIKE '%Via%';
		\end{lstlisting}
		
		% ------ 3 ------
		\item \textbf{Ordini del cliente con id\_cliente = 7}
		\begin{lstlisting}
			SELECT id_ordine, data_ordine, ora_ordine, tipo, stato
			FROM ORDINI
			WHERE id_cliente = 7;
		\end{lstlisting}
		
		% ------ 4 ------
		\item \textbf{Ordini di febbraio 2026 con dati cliente}
		\begin{lstlisting}
			SELECT o.id_ordine, c.cognome, c.nome, o.tipo, o.stato
			FROM ORDINI o
			JOIN CLIENTI c ON c.id_cliente = o.id_cliente
			WHERE o.data_ordine >= '2026-02-01'
			AND o.data_ordine <  '2026-03-01';
		\end{lstlisting}
		
		% ------ 5 ------
		\item \textbf{Clienti che non hanno mai effettuato ordini}
		\begin{lstlisting}
			SELECT c.id_cliente, c.cognome, c.nome
			FROM CLIENTI c
			LEFT JOIN ORDINI o ON o.id_cliente = c.id_cliente
			WHERE o.id_ordine IS NULL;
		\end{lstlisting}
		
		% ------ 6 ------
		\item \textbf{Pizze mai ordinate}
		\begin{lstlisting}
			SELECT p.id_pizza, p.nome, p.categoria
			FROM PIZZE p
			LEFT JOIN RIGHE_ORDINE r ON r.id_pizza = p.id_pizza
			WHERE r.id_pizza IS NULL;
		\end{lstlisting}
		
		% ------ 7 ------
		\item \textbf{Importo totale per ogni ordine}
		\begin{lstlisting}
			SELECT id_ordine,
			SUM(quantita * prezzo_unitario) AS totale
			FROM RIGHE_ORDINE
			GROUP BY id_ordine
			ORDER BY totale DESC;
		\end{lstlisting}
		
		% ------ 8 ------
		\item \textbf{Ordini senza pagamento con esito positivo}
		\begin{lstlisting}
			SELECT o.id_ordine, o.id_cliente,
			o.data_ordine, o.tipo, o.stato
			FROM ORDINI o
			LEFT JOIN PAGAMENTI p
			ON p.id_ordine = o.id_ordine
			AND p.esito = 'OK'
			WHERE p.id_pagamento IS NULL;
		\end{lstlisting}
		
		% ------ 9 ------
		\item \textbf{Spesa totale per cliente (solo pagamenti con esito OK)}
		\begin{lstlisting}
			SELECT c.id_cliente, c.cognome, c.nome,
			COALESCE(SUM(p.importo), 0) AS spesa_totale
			FROM CLIENTI c
			LEFT JOIN ORDINI o    ON o.id_cliente = c.id_cliente
			LEFT JOIN PAGAMENTI p ON p.id_ordine  = o.id_ordine
			AND p.esito = 'OK'
			GROUP BY c.id_cliente, c.cognome, c.nome;
		\end{lstlisting}
		
		% ------ 10 ------
		\item \textbf{Clienti che hanno ordinato tutte le pizze della categoria ``Classiche''}
		
		\textit{Idea:} per ogni cliente, contare le pizze distinte della categoria \texttt{'Classiche'} effettivamente ordinate e confrontarle con il totale delle pizze di quella categoria.
		\begin{lstlisting}
			SELECT c.id_cliente, c.cognome, c.nome
			FROM CLIENTI c
			JOIN ORDINI o       ON o.id_cliente = c.id_cliente
			JOIN RIGHE_ORDINE r ON r.id_ordine  = o.id_ordine
			JOIN PIZZE p        ON p.id_pizza   = r.id_pizza
			WHERE p.categoria = 'Classiche'
			GROUP BY c.id_cliente, c.cognome, c.nome
			HAVING COUNT(DISTINCT p.id_pizza) = (
			SELECT COUNT(id_pizza)
			FROM PIZZE
			WHERE categoria = 'Classiche'
			);
		\end{lstlisting}
		
	\end{enumerate}
	
	% ============================================================
	\section*{Soluzioni -- Parte B (Algebra Relazionale)}
	% ============================================================
	
	\textit{Nota:} si usa l'algebra relazionale estesa con l'operatore di raggruppamento $\gamma$.
	
	\begin{enumerate}
		
		% ------ 1 ------
		\item \textbf{Nomi delle pizze della categoria ``Classiche''}
		\[
		\pi_{\text{nome}}\!\left(
		\sigma_{\text{categoria}=\textquoteleft\text{Classiche}\textquoteright}(\text{PIZZE})
		\right)
		\]
		
		% ------ 2 ------
		\item \textbf{Cognome e nome dei clienti con almeno un ordine}
		\[
		\pi_{\text{cognome},\,\text{nome}}\!\left(
		\text{CLIENTI}
		\bowtie_{\text{CLIENTI.id\_cliente}=\text{ORDINI.id\_cliente}}
		\text{ORDINI}
		\right)
		\]
		
		% ------ 3 ------
		\item \textbf{Identificativi degli ordini consegnati nel 2026}
		\[
		\pi_{\text{id\_ordine}}\!\left(
		\sigma_{\;
			\text{stato}=\textquoteleft\text{consegnato}\textquoteright
			\;\wedge\;
			\textquoteleft\text{2026-01-01}\textquoteright \le \text{data\_ordine}
			< \textquoteleft\text{2027-01-01}\textquoteright
		}(\text{ORDINI})
		\right)
		\]
		
		
		% ------ 5 ------
		\item \textbf{Coppie (cliente, id\_ordine) per ordini di tipo ``domicilio''}
		\[
		\pi_{\text{CLIENTI.id\_cliente},\,\text{id\_ordine}}\!\left(
		\text{CLIENTI}
		\bowtie_{\text{CLIENTI.id\_cliente}=\text{ORDINI.id\_cliente}}
		\sigma_{\text{tipo}=\textquoteleft\text{domicilio}\textquoteright}(\text{ORDINI})
		\right)
		\]
		
	\end{enumerate}
