ESERCIZO 2.1

INNER JOIN ricoveri ON reparti.id_reparto = ricoveri.id_reparto 
WHERE piano IN ( '2', '3') 
GROUP BY nome_reparto;

RIGHT JOIN visite ON pazienti.id_paziente = visite.id_paziente

GROUP BY Nominativo_paziente;

GROUP BY nome_reparto;

ESERCIZIO 4
CREATE TABLE sale(
	id_sala int PRIMARY KEY,
    edificio varchar(50) NOT Null,
    capienza int(11) NOT null
);

CREATE TABLE interventi(
	id_interventi int PRIMARY KEY AUTO_INCREMENT,
    descrizione varchar(500) NOT null,
    data_intervento DATE NOT null,
    id_sala int,
    FOREIGN KEY id_sala REFERENCES sale(id_sala),
    id_medico int,
    FOREIGN KEY id_medico REFERENCES medici(id_medico)
);
